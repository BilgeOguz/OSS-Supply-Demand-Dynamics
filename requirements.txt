#!/usr/bin/env python3
"""Convert pipeline Parquet files to CSV for spreadsheet/tool consumption.

Usage:
    python3 convert_parquet_to_csv.py [--data-dir data/] [--output-dir output/]

Outputs:
    data/multilang_event_indices.parquet → output/raw_events.csv
    data/multilang_health_indices.parquet → output/feature_store.csv
"""

import os
import sys
import argparse

try:
    import pandas as pd
except ImportError:
    print("pandas is required. Install with: pip install pandas")
    sys.exit(1)

try:
    import pyarrow.parquet as pq
except ImportError:
    print("pyarrow is required. Install with: pip install pyarrow")
    sys.exit(1)


def convert_parquet_to_csv(parquet_path, csv_path):
    df = pd.read_parquet(parquet_path)
    df.to_csv(csv_path, index=False)
    print(f"  {parquet_path}  →  {csv_path}  ({len(df)} rows, {len(df.columns)} cols)")


def main():
    parser = argparse.ArgumentParser(description="Convert pipeline Parquet files to CSV")
    parser.add_argument("--data-dir", default="data", help="Directory containing .parquet files")
    parser.add_argument("--output-dir", default="output", help="Directory to write CSV files")
    args = parser.parse_args()

    base_dir = os.path.dirname(os.path.abspath(__file__))
    data_dir = os.path.join(base_dir, args.data_dir)
    output_dir = os.path.join(base_dir, args.output_dir)
    os.makedirs(output_dir, exist_ok=True)

    pairs = [
        ("multilang_event_indices.parquet", "raw_events.csv"),
        ("multilang_health_indices.parquet", "feature_store.csv"),
    ]

    print(f"Converting Parquet → CSV ({os.path.abspath(data_dir)}/)")
    for parquet_name, csv_name in pairs:
        parquet_path = os.path.join(data_dir, parquet_name)
        csv_path = os.path.join(output_dir, csv_name)
        if not os.path.exists(parquet_path):
            print(f"  WARNING: {parquet_path} not found — skipping")
            continue
        try:
            convert_parquet_to_csv(parquet_path, csv_path)
        except Exception as e:
            print(f"  ERROR converting {parquet_name}: {e}")


if __name__ == "__main__":
    main()
